```typescript
import { glob } from 'astro/loaders';
import { defineCollection, z, type ImageFunction } from 'astro:content';

const imageSchema = (image: ImageFunction) =>
    z.object({
        src: image(),
        alt: z.string().optional()
    });

const seoSchema = (image: ImageFunction) =>
    z.object({
        title: z.string().min(5).max(120).optional(),
        description: z.string().min(15).max(160).optional(),
        image: imageSchema(image).optional(),
        pageType: z.enum(['website', 'article']).default('website')
    });

const blog = defineCollection({
    loader: glob({ pattern: '**/*.{md,mdx}', base: './src/content/blog' }),
    schema: ({ image }) =>
        z.object({
            title: z.string(),
            excerpt: z.string().optional(),
            publishDate: z.coerce.date(),
            updatedDate: z.coerce.date().optional(),
            featureImage: imageSchema(image).extend({ caption: z.string().optional() }).optional(),
            categories: z.array(z.string()).optional(),
            isFeatured: z.boolean().default(false),
            seo: seoSchema(image).optional()
        })
});

const pages = defineCollection({
    loader: glob({ pattern: '**/*.{md,mdx}', base: './src/content/pages' }),
    schema: ({ image }) =>
        z.object({
            title: z.string(),
            featureImage: imageSchema(image).extend({ caption: z.string().optional() }).optional(),
            seo: seoSchema(image).optional()
        })
});

export const collections = { blog, pages };
```

Install gray-matter: `npm install gray-matter`

Create `scripts/migrate.mjs` -- a Node.js script that migrates all WordPress export content to Astro content collections. The script should:

**CLI interface:**
- `node scripts/migrate.mjs --type=posts` (migrate blog posts)
- `node scripts/migrate.mjs --type=pages` (migrate pages)
- `node scripts/migrate.mjs --type=videos` (migrate videos)
- `node scripts/migrate.mjs --type=all` (migrate everything)
- `node scripts/migrate.mjs --dry-run --type=all` (show what would be done, no file writes)

**Blog post migration (`content-export/posts/*.md` -> `src/content/blog/{slug}/index.md`):**

1. Frontmatter transformation:
   - `date` -> `publishDate`
   - `featured_image` (URL string) -> `featureImage: { src: ./images/{filename}, alt: "" }`
   - `categories` array: if items are `{name, url}` objects, extract `.name`; if already strings, use as-is
   - Auto-generate `excerpt` from first non-heading paragraph of body, truncated to 160 chars
   - Remove `author` field (single-author blog, not in schema)
   - Remove any other fields not in the blog schema

2. Content cleanup (apply to body text):
   - Strip the first `# Title` line (duplicate of frontmatter title)
   - Remove `Estimated reading time: X minutes` lines
   - Remove `Code language: ...` suffixes that appear after code blocks (pattern: lines starting with `Code language:`)
   - Convert absolute internal links: `https://joost.blog/slug/` -> `/slug/`
   - Fix HTML entities: `&amp;` -> `&`
   - Strip WordPress block comments: `<!-- wp:anything -->` and `<!-- /wp:anything -->`
   - Handle `\[wp_playground ...\]\[/wp_playground\]` escaped shortcodes: remove entirely (only in embedded-playground page)
   - Simplify linked images `[![alt](thumb-url)](full-url)` -> `![alt](full-url)` (keep the full-size image URL)
   - Trim excessive blank lines (3+ consecutive newlines -> 2)

3. Image handling:
   - Extract all image URLs from: `featured_image` frontmatter, inline `![alt](url)` markdown images
   - Download each image to `src/content/blog/{slug}/images/{original-filename}`
   - Use concurrent downloads (max 5 at a time) with retry on failure
   - Update frontmatter `featureImage.src` to `./images/{filename}`
   - Update inline image markdown to `./images/{filename}`
   - Skip download if file already exists (idempotent reruns)
   - Log warnings for failed downloads but don't abort the entire migration

**Page migration (`content-export/pages/*.md` -> `src/content/pages/{slug}.md`):**

1. Frontmatter: keep only `title` (the pages schema only requires title)
2. Content cleanup: same body cleanup as blog posts
3. Special handling:
   - `blog.md`: SKIP this file (Astro generates blog listing from routing)
   - `contact-me.md`: Strip Gravity Forms HTML remnants (any `<div class="gf_...">` blocks, honeypot fields, form markup). Leave the text content.
   - `embedded-playground.md`: Remove the escaped `\[wp_playground\]` shortcode block entirely, leave surrounding text
   - `clicky.md`: This is a subpage of plugins. Output to `src/content/pages/clicky.md` (flat structure -- routing handles nesting)
   - `videos.md`: Migrate as a page if it has meaningful content beyond just a listing
4. Pages do NOT have co-located images. Any inline images in pages should be downloaded to `src/content/pages/images/` and referenced as `./images/{filename}`. Check if any pages actually have inline images first.

**Video migration (`content-export/videos/*.md` -> `src/content/videos/{slug}.md`):**

1. Frontmatter transformation:
   - `date` -> `publishDate`
   - Remove `author`, `source_url`, `note` fields
   - Add YouTube metadata from this hardcoded lookup table:

   ```javascript
   const videoMetadata = {
     'wordpress-seo-in-2019': { youtubeId: 'pS-OhpUMC10', duration: 'PT44M14S' },
     '10-years-of-yoast': { youtubeId: '27_iFw_W-HE', duration: 'PT33M46S' },
     'why-timestamping-will-be-good-for-seo': { youtubeId: 'r6T-WuqjGI0', duration: 'PT30M8S' },
     'the-victory-of-the-commons': { videoUrl: 'https://wordpress.tv/2013/12/12/joost-de-valk-the-victory-of-the-commons/' },
     'the-business-of-open-source': { videoUrl: 'https://wordpress.tv/2018/12/27/joost-de-valk-the-business-of-open-source/' },
     'growing-a-multi-million-dollar-business-with-a-plugin': { videoUrl: 'https://wordpress.tv/2020/06/23/joost-de-valk-growing-a-multi-million-dollar-business-with-a-wordpress-plugin/' },
     'sustainable-open-source-is-the-future': { videoUrl: 'https://wordpress.tv/2024/07/03/joost-de-valk-sustainable-open-source-is-the-future/' },
     'improve-the-environment-start-with-your-website': { videoUrl: 'https://wordpress.tv/2022/11/17/joost-de-valk-improve-the-environment-start-with-your-website/' },
   };
   ```

   - If categories exist and are objects, flatten to strings (same as posts)
   - Remove categories field entirely from video frontmatter (not in videos schema)

2. Content cleanup: strip H1 title, same text cleanup as posts
3. Clean up body: remove any raw HTML embed markup, keep only the descriptive text

**Logging:**
- Print a summary at the end: X posts migrated, Y pages migrated, Z videos migrated, N images downloaded, M warnings
- Log each file as it's processed: `[POST] tailwind-paradox -> src/content/blog/tailwind-paradox/index.md`
- Log each image download: `[IMG] downloaded https://... -> src/content/blog/slug/images/file.ext`
- Log warnings for: failed image downloads, missing expected fields, skipped files

**Error handling:**
- Failed image downloads: warn and continue (don't abort)
- Malformed frontmatter: warn and skip file
- File already exists in target: overwrite (script should be idempotent)
